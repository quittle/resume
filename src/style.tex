\pgfkeys {
    /customHeader/.is family, /customHeader,
    firstName/.estore in = \customHeader@firstName,
    lastName/.estore in = \customHeader@lastName,
    email/.estore in = \customHeader@email,
    phone/.estore in = \customHeader@phone,
    resumeLink/.estore in = \customHeader@resumeLink,
    website/.estore in = \customHeader@website,
    websiteLink/.estore in = \customHeader@websiteLink,
    githubUser/.estore in = \customHeader@githubUser,
}
\newcommand {\customHeader} [7] [] {
    \pgfkeys {/customHeader,#1}
    \parbox {0.3\textwidth} {
        \vspace {0.3cm}
        \href {\customHeader@resumeLink} {\Huge {\color {slateblue} \textbf {\customHeader@firstName\ \customHeader@lastName} }} \\ % Header section name and color
    }
    \hfill
    \parbox {\textwidth} {
        \begin {rtabbing}
            \hspace{2.6cm} \= \customHeader@email \= \hspace{0.3cm} \= \hfill {\raggedleft \href {\customHeader@websiteLink} {\customHeader@website}} \\
            \customHeader@phone \> \> \> \href {https://github.com/\customHeader@githubUser} {github.com/\customHeader@githubUser}
        \end {rtabbing}
    }
    \rule {\textwidth} {0.5mm} % Rule under the header
}
