\pgfkeys {
    /title/.is family, /title,
    first/.estore in = \title@first,
    middle/.estore in = \title@middle,
    last/.estore in = \title@last,
}
\renewcommand {\title} [3] [] {
    \pgfkeys{/title,#1}
    {\huge {\color {slateblue} \textbf {\title@first\ \title@last} }} \\ % Header section name and color
    \rule {\textwidth} {0.5mm} % Rule under the header
}

\pgfkeys {
    /contactgroup/.is family, /contactgroup,
    city/.estore in = \contactgroup@city,
    state/.estore in = \contactgroup@state,
    zipcode/.estore in = \contactgroup@zipcode,
    email/.estore in = \contactgroup@email,
    phone/.estore in = \contactgroup@phone
}
\newcommand {\contactgroup} [5] [] {
    \pgfkeys{/contactgroup,#1}

    \parbox {0.5\textwidth} { % First block
        \begin {tabbing}
            \hspace{3cm} \= \hspace{4cm} \= \kill % Spacing within the block
            {\bf Location} \> \contactgroup@city, \contactgroup@state \\
            \\ % Blank line
        \end {tabbing}
    }
    \hfill % Horizontal space between the two blocks
    \parbox {0.5\textwidth} { % Second block
        \begin {tabbing}
            \hspace{3cm} \= \hspace{4cm} \= \kill % Spacing within the block
            {\bf Email} \> \email {\contactgroup@email} \\ % Email address
            {\bf Cell}  \> \contactgroup@phone \\ % Mobile phone
        \end {tabbing}
    }
}
